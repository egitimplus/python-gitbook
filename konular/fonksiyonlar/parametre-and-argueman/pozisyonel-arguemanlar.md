```python
def fruit_juicer(fruit): # name parametre
    print(f'I am {fruit} juicer')

```

```python
def fruit_juicer(fruit1, fruit2):
    print(f'I am {fruit1} and {fruit2} juicer')

fruit_juicer('orange', 'apple') 
# I am orange and apple juicer
```

```python
def sum(a, b):
    print(a + b)

sum(3, 5) 
# 8
```
